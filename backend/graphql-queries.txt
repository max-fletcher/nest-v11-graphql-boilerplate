### BOOKS ###

Get Books:
query {
  books {
    id
    title
    author {
      id
      name
    }
  }
}

Get Single Book:
query GetBook($id: ID!) {
  book(id: $id) {
    id
    title
    author {
      id
      name
    }
  }
}
// variables panel:
{
 "id": "cmraoynyn0002rovho96s1c3f"
}

Create Book:
mutation {
  createBook(createAuthorInput: { title: "The Cat In The Hat", authorId: "cmrao64d10000rovhg301pb2p" }) {
    id
    title
    author {
      id
      name
    }
  }
}


Update book:
mutation {
  updateBook(updateAuthorInput: { id: "cmrs1ipf20000skvhy3i6g1f2", title: "The Cat In The Hat2", authorId: "cmrao64d10000rovhg301pb2p" }) {
    id
    title
    author {
      id
      name
    }
  }
}

Delete book:
mutation {
  removeBook(id: "cmrs1ipf20000skvhy3i6g1f2")
}
